import {Component, OnInit} from '@angular/core';
import {pickerI18n} from '../../app/talon/talon.component';
import {Patient, PatientDocument} from './../patient';
import {MatDialog, MatDialogConfig} from '@angular/material';
import {PatientDocumentModalComponent} from './patient-document-modal/patient-document-modal.component';
import {MockService} from '../service/mock.service';
import {ActivatedRoute} from '@angular/router';
import {PatientService} from '../service/patient.service';
import {PatientHistoryModalComponent} from './patient-history-modal/patient-history-modal.component';
import {PatientUnionModalComponent} from './patient-union-modal/patient-union-modal.component';

@Component({
  selector: 'app-patient-card',
  templateUrl: './patient-card.component.html',
  styleUrls: ['./patient-card.component.sass']
})
export class PatientCardComponent implements OnInit {
  pickerI18n = pickerI18n;
  patientInform = new Patient();
  documents: PatientDocument[];
  dialogConfig = new MatDialogConfig();
  showEnableInfo = false;

  constructor(public dialog: MatDialog, private mock: MockService, private route: ActivatedRoute, private patientApi: PatientService) {

  }

  openPatientDoc() {
    this.dialog.open(PatientDocumentModalComponent, this.dialogConfig);
  }

  openPatientUnion() {
    this.dialog.open(PatientUnionModalComponent, this.dialogConfig);
  }

  openPatientHistory() {
    if (this.patientInform.id) {
      this.dialogConfig.data = {id: this.patientInform.id};
    }
    this.dialog.open(PatientHistoryModalComponent, this.dialogConfig);
  }

  ngOnInit() {
    this.dialogConfig.disableClose = false;
    this.dialogConfig.autoFocus = true;
    this.dialogConfig.hasBackdrop = true;
    this.dialogConfig.backdropClass = '';
    this.dialogConfig.width = '55%';
    this.dialogConfig.height = '70%';
    const params = this.route.snapshot.paramMap;
    this.documents = this.mock.state;
    if (params.has('id')) {
      this.getPatenInfo(Number(params.get('id')));
    }
  }

  deleteState(index: number) {
    if (index > -1) {
      this.documents.splice(index, 1);
    }
  }

  getPatenInfo(id: number) {
    this.patientApi.getPatient(id).subscribe(
      (response) => {
        console.log(response);
        this.showEnableInfo = true;
        this.patientInform = response;
      }
    );
  }

  updatePatientInfo() {
    this.patientApi.updatePatient(this.patientInform.id, this.patientInform).subscribe(
      response => {
        console.log('Uraa');
      }
    );
  }
}
