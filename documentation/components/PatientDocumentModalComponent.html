import {Component, OnInit} from '@angular/core';
import {MockService} from '../../service/mock.service';
import {PatientDocument} from '../../patient';
import {MatDialog, MatDialogRef} from "@angular/material";

@Component({
  selector: 'app-patient-document-modal',
  templateUrl: './patient-document-modal.component.html',
  styleUrls: ['./patient-document-modal.component.sass']
})

/**
 * Модальное окно для заполенение пасспорта пациента
 */
export class PatientDocumentModalComponent implements OnInit {

  constructor(private mock: MockService, private matDialogRef: MatDialogRef<PatientDocumentModalComponent>) {
  }

  ngOnInit() {
  }

  /**
   *  Временное хранение
   */
  saveData() {
    const data: PatientDocument = {
      id: 0,
      dateDocument: 0,
      documentNumber: 0,
      documentSeries: 0,
      postDocument: 'seria',
      documentType : 'Пасспорт'
    };
    this.mock.state.push(data);
    this.matDialogRef.close();
  }
}
